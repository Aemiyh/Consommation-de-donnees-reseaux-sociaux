    var margin = {

    var height = 470 - margin.top - margin.bottom;

    var chart1 = d3.select("body").append("svg")
        .attr("width", width + margin.left + margin.right)
        .attr("height", height + margin.top + margin.bottom)
      .append("g")
        .attr("transform", "translate(" + margin.left + "," + margin.top + ")");
   	
		console.log(height)
    var x = d3.scaleLinear().range([0, width]);

    var y = d3.scaleBand().range([height - 50, 0]);

    var xAxis = d3.axisTop(x).ticks(10, "%");
    var yAxis = d3.axisLeft(y);

    var bandwidth = 0;
    
    // Premiere page
    d3.csv("donnees_App_Mobile.csv")
      .then(function(data) 
        {
      		updateChart(chart1,data,1);
    		});// fin load
		

    

    
    
    
   
    //Ajout chart
    function updateChart(svg,data,int){
      //svg.selectAll("*").remove();
     //Inverser l'ordre des lignes
      data.sort(function(a, b) {
        return d3.descending(a.application, b.application)
      });
      var maxFrequency = d3.max(data, function(d) { return d.frequency; });
      
      x.domain([0, 1]);

      y.domain(data.map(function(d,i) { console.log("");return i; }))
        .paddingInner(0.1);

      svg.append("g")
        .attr("class", "xaxis")
        .attr("transform", "translate(0," + -2 + ")")
        .call(xAxis);

      var g = svg.append("g"); 
			
      console.log(y.bandwidth())
      g.attr("class", "y axis")
        .attr("transform", `translate(${-10},${y.bandwidth()})`);

      g.selectAll("text")
        .data(data)
        .enter().append("text")
        .text(function(d) { return d.application.toUpperCase(); })
        .attr("class", "label")
        .attr("x", 0)
        .attr("y", function(d,i) { return y(i); })
        .attr("dy", -2);

      svg.selectAll(".bar"+int)
        .data(data)
        .enter().append("rect")
        .attr("class", "bar"+int)
        .attr("x", -2)
        .attr("height", function(d) {
        bandwidth = y.bandwidth();
        return bandwidth;
      })
        .attr("y", function(d, i) { return y(i); })
        .attr("width", function(d) { return x(d.somme); })
        .on("mouseover", function(d) {
        //do
        })
        .on("mouseout", function(d) {
        //do
        }); 
     } 
    

  </script>
